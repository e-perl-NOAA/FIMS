186 virtual double evaluate_mean(double spawners, double ssbzero) {
187 fims_popdy::SRBevertonHolt<double> BevHolt;
188 BevHolt.logit_steep.resize(1);
189 BevHolt.logit_steep[0] = this->logit_steep[0].initial_value_m;
190 if (this->logit_steep[0].initial_value_m == 1.0) {
191 warning(
192 "Steepness is subject to a logit transformation. "
193 "Fixing it at 1.0 is not currently possible.");

196 BevHolt.log_rzero[0] = this->log_rzero[0].initial_value_m;
197
