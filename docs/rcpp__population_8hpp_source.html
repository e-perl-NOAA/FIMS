262 if (this->finalized) {
263 // log warning that finalize has been called more than once.
264 FIMS_WARNING_LOG("Population " + fims::to_string(this->id) +

278 info->populations[this->id];
279 it = info->populations.find(this->id);
280 if (it == info->populations.end()) {
281 FIMS_WARNING_LOG("Population " + fims::to_string(this->id) +
282 " not found in Information.");
283 return;
284 } else {
285 for (size_t i = 0; i < this->log_M.size(); i++) {
286 if (this->log_M[i].estimation_type_m.get() == "constant") {
287 this->log_M[i].final_value_m = this->log_M[i].initial_value_m;
288 } else {
289 this->log_M[i].final_value_m = pop->log_M[i];
290 }
291 }
292
293 for (size_t i = 0; i < this->log_init_naa.size(); i++) {

295 this->log_init_naa[i].final_value_m =
296 this->log_init_naa[i].initial_value_m;
297 } else {
298 this->log_init_naa[i].final_value_m = pop->log_init_naa[i];
299 }
300 }
301
302 // set the derived quantities
303 this->derived_naa = Rcpp::NumericVector(pop->numbers_at_age.size());
304 this->derived_ssb = Rcpp::NumericVector(pop->spawning_biomass.size());
305 this->derived_biomass = Rcpp::NumericVector(pop->biomass.size());
306 this->derived_recruitment =

308
309 // set naa from Information/
310 for (R_xlen_t i = 0; i < this->derived_naa.size(); i++) {
311 this->derived_naa[i] = pop->numbers_at_age[i];
312 }
313
314 // set ssb from Information/
315 for (R_xlen_t i = 0; i < this->derived_ssb.size(); i++) {
316 this->derived_ssb[i] = pop->spawning_biomass[i];
317 }
318
319 // set biomass from Information

321 this->derived_biomass[i] = pop->biomass[i];
322 }
323
324 // set recruitment from Information/

326 this->derived_recruitment[i] = pop->expected_recruitment[i];

438 population->nages = this->nages.get();

451 population->recruitment_id = this->recruitment_id.get();
452 population->maturity_id = this->maturity_id.get();
453 population->log_M.resize(this->log_M.size());
454 population->log_init_naa.resize(this->log_init_naa.size());

456 population->log_M[i] = this->log_M[i].initial_value_m;

458 ss.str("");

461 info->RegisterParameter(population->log_M[i]);
462 }
463 if (this->log_M[i].estimation_type_m.get() == "random_effects") {
464 ss.str("");
465 ss << "Population." << this->id << ".log_M." << this->log_M[i].id_m;
466 info->RegisterRandomEffectName(ss.str());
467 info->RegisterRandomEffect(population->log_M[i]);
468 }
469 }
470 info->variable_map[this->log_M.id_m] = &(population)->log_M;
471
472 for (size_t i = 0; i < log_init_naa.size(); i++) {
473 population->log_init_naa[i] = this->log_init_naa[i].initial_value_m;

475 ss.str("");

478 info->RegisterParameterName(ss.str());
479 info->RegisterParameter(population->log_init_naa[i]);

481 if (this->log_init_naa[i].estimation_type_m.get() == "random_effects") {
482 ss.str("");
483 ss << "Population." << this->id << ".log_init_naa."

485 info->RegisterRandomEffectName(ss.str());
486 info->RegisterRandomEffect(population->log_init_naa[i]);
487 }
488 }
489 info->variable_map[this->log_init_naa.id_m] = &(population)->log_init_naa;
490 for (int i = 0; i < ages.size(); i++) {
491 population->ages[i] = this->ages[i];
492 }
493
494 population->numbers_at_age.resize((nyears + 1) * nages);
495 info->variable_map[this->numbers_at_age.id_m] =
496 &(population)->numbers_at_age;
497
498 // add to Information
499 info->populations[population->id] = population;
500
501 return true;
