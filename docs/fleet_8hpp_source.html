245 void evaluate_age_comp() {
246 for (size_t y = 0; y < this->nyears; y++) {
247 Type sum = static_cast<Type>(0.0);
248 Type sum_obs = static_cast<Type>(0.0);
249 // robust_add is a small value to add to expected compostion
250 // proportions at age to stabilize likelihood calculations
251 // when the expected proportions are close to zero.
252 // Type robust_add = static_cast<Type>(0.0); // zeroed out before testing
253 // 0.0001;
254 // sum robust is used to calculate the total sum of robust
255 // additions to ensure that proportions sum to 1.
256 // Type robust_sum = static_cast<Type>(1.0);
257
258 for (size_t a = 0; a < this->nages; a++) {
259 size_t i_age_year = y * this->nages + a;
260 // Here we have a check to determine if the age comp
261 // should be calculated from the retained landings or
262 // the total population. These values are slightly different.
263 // In the future this will have more impact as we implement
264 // timing rather than everything occuring at the start of
265 // the year.
266 if (this->fleet_observed_landings_data_id_m == -999) {
267 this->agecomp_expected[i_age_year] =
268 this->index_numbers_at_age[i_age_year];
269 } else {
270 this->agecomp_expected[i_age_year] =
271 this->landings_numbers_at_age[i_age_year];
272 }
273 sum += this->agecomp_expected[i_age_year];
274 // robust_sum -= robust_add;
275
276 // This sums over the observed age composition data so that
277 // the expected age composition can be rescaled to match the
278 // total number observed. The check for na values should not
279 // be needed as individual years should not have missing data.
280 // This is need to be re-explored if/when we modify FIMS to
281 // allow for composition bins that do not match the population
282 // bins.
283 if (this->fleet_observed_agecomp_data_id_m != -999) {
284 if (this->observed_agecomp_data->at(i_age_year) !=
285 this->observed_agecomp_data->na_value) {
286 sum_obs += this->observed_agecomp_data->at(i_age_year);
287 }
288 }
289 }
290 for (size_t a = 0; a < this->nages; a++) {
291 size_t i_age_year = y * this->nages + a;
292 this->agecomp_proportion[i_age_year] =
293 this->agecomp_expected[i_age_year] / sum;
294 // robust_add + robust_sum * this->agecomp_expected[i_age_year] / sum;
295
296 if (fleet_observed_agecomp_data_id_m != -999) {
297 this->agecomp_expected[i_age_year] =
298 this->agecomp_proportion[i_age_year] * sum_obs;
299 }
300 }
301 }

307 void evaluate_length_comp() {
308 if (this->nlengths > 0) {
309 for (size_t y = 0; y < this->nyears; y++) {
310 Type sum = static_cast<Type>(0.0);
311 Type sum_obs = static_cast<Type>(0.0);
312 // robust_add is a small value to add to expected compostion
313 // proportions at age to stabilize likelihood calculations
314 // when the expected proportions are close to zero.
315 // Type robust_add = static_cast<Type>(0.0); // 0.0001; zeroed out
316 // before testing
317 // sum robust is used to calculate the total sum of robust
318 // additions to ensure that proportions sum to 1.
319 // Type robust_sum = static_cast<Type>(1.0);
320 for (size_t l = 0; l < this->nlengths; l++) {
321 size_t i_length_year = y * this->nlengths + l;
322 for (size_t a = 0; a < this->nages; a++) {
323 size_t i_age_year = y * this->nages + a;
324 size_t i_length_age = a * this->nlengths + l;
325 this->lengthcomp_expected[i_length_year] +=
326 this->agecomp_expected[i_age_year] *
327 this->age_to_length_conversion[i_length_age];
328
329 this->landings_numbers_at_length[i_length_year] +=
330 this->landings_numbers_at_age[i_age_year] *
331 this->age_to_length_conversion[i_length_age];
332
333 this->index_numbers_at_length[i_length_year] +=
334 this->index_numbers_at_age[i_age_year] *
335 this->age_to_length_conversion[i_length_age];
336 }
337
338 sum += this->lengthcomp_expected[i_length_year];

341 if (this->fleet_observed_lengthcomp_data_id_m != -999) {
