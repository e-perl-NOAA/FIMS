258template <class Type>
259inline const Type double_logistic(const Type &inflection_point_asc,
260 const Type &slope_asc,
261 const Type &inflection_point_desc,
262 const Type &slope_desc, const Type &x) {
263 return (static_cast<Type>(1.0)) /
264 (static_cast<Type>(1.0) +
265 exp(Type(-1.0) * slope_asc * (x - inflection_point_asc))) *
266 (static_cast<Type>(1.0) -
267 (static_cast<Type>(1.0)) /
268 (static_cast<Type>(1.0) +
269 exp(Type(-1.0) * slope_desc * (x - inflection_point_desc))));
270}
