337 virtual void Prepare() {
338 for (size_t p = 0; p < this->populations.size(); p++) {
339 this->populations[p]->Prepare();
340
341 auto derived_quantities =
342 this->population_derived_quantities[this->populations[p]->GetId()];
343
344 typename fims_popdy::Population<Type>::derived_quantities_iterator it;
345 for (it = derived_quantities.begin(); it != derived_quantities.end();
346 it++) {
347 fims::Vector<Type> &dq = (*it).second;
348 this->ResetVector(dq);
349 }
350 }
351
352 for (size_t p = 0; p < this->populations.size(); p++) {
353 std::shared_ptr<fims_popdy::Population<Type>> &population =
354 this->populations[p];
355 std::map<std::string, fims::Vector<Type>> &derived_quantities =
356 this->population_derived_quantities[population->GetId()];
357
358 // Transformation Section
359 for (size_t age = 0; age < population->nages; age++) {
360 for (size_t year = 0; year < population->nyears; year++) {
361 size_t i_age_year = age * population->nyears + year;
362 population->M[i_age_year] =
363 fims_math::exp(population->log_M[i_age_year]);
364 // mortality_F is a fims::Vector and therefore needs to be filled
365 // within a loop
366 derived_quantities["mortality_F"][i_age_year] = 0.0;
367 derived_quantities["weight_at_age"][i_age_year] =
368 population->growth->evaluate(population->ages[age]);
369 }
370 }
371 }
372
373 for (fleet_iterator fit = this->fleets.begin(); fit != this->fleets.end();
374 ++fit) {
375 std::shared_ptr<fims_popdy::Fleet<Type>> &fleet = (*fit).second;
376 std::map<std::string, fims::Vector<Type>> &derived_quantities =
377 this->fleet_derived_quantities[fleet->GetId()];
378 typename fims_popdy::Population<Type>::derived_quantities_iterator it;
379 for (it = derived_quantities.begin(); it != derived_quantities.end();
380 it++) {
381 fims::Vector<Type> &dq = (*it).second;
382 this->ResetVector(dq);
383 }
384
385 for (size_t i_length_age = 0;
386 i_length_age < fleet->age_to_length_conversion.size();
387 i_length_age++) {
388 this->fleet_derived_quantities[fleet->GetId()]
389 ["age_to_length_conversion"]
390 [i_length_age] =
391 fleet->age_to_length_conversion[i_length_age];
392 }
393 }

517 void CalculateMortality(
518 std::shared_ptr<fims_popdy::Population<Type>> &population,
519 size_t i_age_year, size_t year, size_t age) {
520 for (size_t fleet_ = 0; fleet_ < population->nfleets; fleet_++) {
521 // evaluate is a member function of the selectivity class
522 Type s = population->fleets[fleet_]->selectivity->evaluate(
523 population->ages[age]);
524
525 this->population_derived_quantities[population->GetId()]["mortality_F"]
526 [i_age_year] +=
527 population->fleets[fleet_]->Fmort[year] * s;
528
529 this->population_derived_quantities[population->GetId()]
530 ["sum_selectivity"][i_age_year] += s;
531 }
532 this->population_derived_quantities[population->GetId()]["mortality_Z"]
533 [i_age_year] =
534 population->M[i_age_year] +
535 this->population_derived_quantities[population->GetId()]["mortality_F"]
536 [i_age_year];
537 }

635 Type CalculateSBPR0(
636 std::shared_ptr<fims_popdy::Population<Type>> &population) {
637 std::vector<Type> numbers_spr(population->nages, 1.0);
638 Type phi_0 = 0.0;
639 phi_0 +=
640 numbers_spr[0] * population->proportion_female[0] *
641 this->population_derived_quantities[population->GetId()]
642 ["proportion_mature_at_age"][0] *
643 population->growth->evaluate(population->ages[0]);
644 for (size_t a = 1; a < (population->nages - 1); a++) {
645 numbers_spr[a] = numbers_spr[a - 1] * fims_math::exp(-population->M[a]);
646 phi_0 +=
647 numbers_spr[a] * population->proportion_female[a] *
648 this->population_derived_quantities[population->GetId()]
649 ["proportion_mature_at_age"][a] *
650 population->growth->evaluate(population->ages[a]);
651 }
652
653 numbers_spr[population->nages - 1] =
654 (numbers_spr[population->nages - 2] *
655 fims_math::exp(-population->M[population->nages - 2])) /
656 (1 - fims_math::exp(-population->M[population->nages - 1]));
657 phi_0 +=
658 numbers_spr[population->nages - 1] *
659 population->proportion_female[population->nages - 1] *
660 this->population_derived_quantities[population->GetId()]
661 ["proportion_mature_at_age"]
662 [population->nages - 1] *
663 population->growth->evaluate(population->ages[population->nages - 1]);
664
665 return phi_0;
666 }

672 void CalculateRecruitment(
673 std::shared_ptr<fims_popdy::Population<Type>> &population,
674 size_t i_age_year, size_t year, size_t i_dev) {
675 Type phi0 = CalculateSBPR0(population);
676
677 if (i_dev == population->nyears) {
678 this->population_derived_quantities[population->GetId()]["numbers_at_age"]
679 [i_age_year] =
680 population->recruitment->evaluate_mean(
681 this->population_derived_quantities[population->GetId()]
682 ["spawning_biomass"][year - 1],
683 phi0);
684 /*the final year of the time series has no data to inform recruitment
685 devs, so this value is set to the mean recruitment.*/
686 } else {
687 // Why are we using evaluate_mean, how come a virtual function was
688 // changed? AMH: there are now two virtual functions: evaluate_mean and
689 // evaluate_process (see below)
690 population->recruitment->log_expected_recruitment[year - 1] =
691 fims_math::log(population->recruitment->evaluate_mean(
692 this->population_derived_quantities[population->GetId()]
693 ["spawning_biomass"][year - 1],
694 phi0));
695
696 this->population_derived_quantities[population->GetId()]["numbers_at_age"]
697 [i_age_year] = fims_math::exp(
698 population->recruitment->process->evaluate_process(year - 1));
699 }
700
701 this->population_derived_quantities[population->GetId()]
702 ["expected_recruitment"][year] =
703 this->population_derived_quantities[population->GetId()]
704 ["numbers_at_age"][i_age_year];
705 }

1201 Performs parameters transformations
1202 Sets recruitment deviations to mean 0.
1203 */
1204 Prepare();
1205 /*
1206 start at year=0, age=0;
1207 here year 0 is the estimated initial population structure and age 0 are

1209 code for initial structure and recruitment 0 loops. Could also have started
1210 loops at 1 with initial structure and recruitment setup outside the loops.
1211
1212 year loop is extended to <= nyears because SSB is calculted as the start of
1213 the year value and by extending one extra year we get estimates of the

1224 this->populations[p];
1225
1226 // CAAPopulationProxy<Type>& population = this->populations_proxies[p];
1227
1228 for (size_t y = 0; y <= population->nyears; y++) {
1229 for (size_t a = 0; a < population->nages; a++) {
1230 /*
1231 index naming defines the dimensional folding structure
1232 i.e. i_age_year is referencing folding over years and ages.
1233 */

1235 /*
1236 Mortality rates are not estimated in the final year which is
1237 used to show expected population structure at the end of the model

1244 */

1247 First thing we need is total mortality aggregated across all fleets
