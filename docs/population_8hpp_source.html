335 Type CalculateSBPR0() {
336 std::vector<Type> numbers_spr(this->nages, 1.0);
337 Type phi_0 = static_cast<Type>(0.0);
338 phi_0 += numbers_spr[0] * this->proportion_female[0] *
339 this->proportion_mature_at_age[0] *
340 this->growth->evaluate(ages[0]);
341 for (size_t a = 1; a < (this->nages - 1); a++) {
342 numbers_spr[a] = numbers_spr[a - 1] * fims_math::exp(-this->M[a]);
343 phi_0 += numbers_spr[a] * this->proportion_female[a] *
344 this->proportion_mature_at_age[a] *
345 this->growth->evaluate(ages[a]);
346 }
347
348 numbers_spr[this->nages - 1] =
349 (numbers_spr[nages - 2] * fims_math::exp(-this->M[nages - 2])) /
350 (1 - fims_math::exp(-this->M[this->nages - 1]));
351 phi_0 += numbers_spr[this->nages - 1] *
352 this->proportion_female[this->nages - 1] *
353 this->proportion_mature_at_age[this->nages - 1] *
354 this->growth->evaluate(ages[this->nages - 1]);
355 return phi_0;
356 }

365 void CalculateRecruitment(size_t i_age_year, size_t year, size_t i_dev) {
366 Type phi0 = CalculateSBPR0();
367
368 if (i_dev == this->nyears) {
369 this->numbers_at_age[i_age_year] = this->recruitment->evaluate_mean(
370 this->spawning_biomass[year - 1], phi0);
371 /*the final year of the time series has no data to inform recruitment
372 devs, so this value is set to the mean recruitment.*/
373 } else {
374 recruitment->log_expected_recruitment[year - 1] =
375 fims_math::log(this->recruitment->evaluate_mean(
376 this->spawning_biomass[year - 1], phi0));
377 /*the log_recruit_err vector does not include a value for year == 0
378 and is of length nyears - 1 where the first position of the vector
379 corresponds to the second year of the time series. The first year is
380 informed by the init_naa parameter*/
381 this->numbers_at_age[i_age_year] =
382 fims_math::exp(recruitment->process->evaluate_process(year - 1));
383 }
384 this->expected_recruitment[year] = this->numbers_at_age[i_age_year];

518 void Evaluate() {
519 /*
520 Sets derived vectors to zero
521 Performs parameters transformations
522 Sets recruitment deviations to mean 0.
523 */
524 Prepare();
525 /*
526 start at year=0, age=0;
527 here year 0 is the estimated initial population structure and age 0 are
528 recruits loops start at zero with if statements inside to specify unique
529 code for initial structure and recruitment 0 loops. Could also have started
530 loops at 1 with initial structure and recruitment setup outside the loops.
531
532 year loop is extended to <= nyears because SSB is calculted as the start of
533 the year value and by extending one extra year we get estimates of the
534 population structure at the end of the final year. An alternative approach
535 would be to keep initial numbers at age in it's own vector and each year to
536 include the population structure at the end of the year. This is likely a
537 null point given that we are planning to modify to an event/stanza based
538 structure in later milestones which will elimitate this confusion by
539 explicity referencing the exact date (or period of averaging) at which any
540 calculation or output is being made.
541 */
542 for (size_t y = 0; y <= this->nyears; y++) {
543 for (size_t a = 0; a < this->nages; a++) {
544 /*
545 index naming defines the dimensional folding structure
546 i.e. i_age_year is referencing folding over years and ages.
547 */
548 size_t i_age_year = y * this->nages + a;
549 /*
550 Mortality rates are not estimated in the final year which is
551 used to show expected population structure at the end of the model
552 period. This is because biomass in year i represents biomass at the
553 start of the year. Should we add complexity to track more values such
554 as start, mid, and end biomass in all years where, start biomass=end
555 biomass of the previous year? Referenced above, this is probably not
556 worth exploring as later milestone changes will eliminate this
557 confusion.
558 */
559 if (y < this->nyears) {
560 /*
561 First thing we need is total mortality aggregated across all fleets
562 to inform the subsequent landings and change in numbers at age
563 calculations. This is only calculated for years < nyears as these are
564 the model estimated years with data. The year loop extends to
565 y=nyears so that population numbers at age and SSB can be calculated
566 at the end of the last year of the model
567 */
568 CalculateMortality(i_age_year, y, a);
569 }
570 CalculateMaturityAA(i_age_year, a);
571 /* if statements needed because some quantities are only needed
572 for the first year and/or age, so these steps are included here.
573 */
574 if (y == 0) {
575 // Initial numbers at age is a user input or estimated parameter
576 // vector.
577 CalculateInitialNumbersAA(i_age_year, a);
578
579 if (a == 0) {
580 this->unfished_numbers_at_age[i_age_year] =
581 fims_math::exp(this->recruitment->log_rzero[0]);
582 } else {
583 CalculateUnfishedNumbersAA(i_age_year, a - 1, a);
584 }
585
586 /*
587 Fished and unfished biomass vectors are summing biomass at
588 age across ages.
589 */
590
591 CalculateBiomass(i_age_year, y, a);
592
593 CalculateUnfishedBiomass(i_age_year, y, a);
594
595 /*
596 Fished and unfished spawning biomass vectors are summing biomass at
597 age across ages to allow calculation of recruitment in the next year.
598 */
599
600 CalculateSpawningBiomass(i_age_year, y, a);
601
602 CalculateUnfishedSpawningBiomass(i_age_year, y, a);
603
604 /*
605 Expected recruitment in year 0 is numbers at age 0 in year 0.
606 */
607
608 this->expected_recruitment[i_age_year] =
609 this->numbers_at_age[i_age_year];
610
611 } else {
612 if (a == 0) {
613 // Set the nrecruits for age a=0 year y (use pointers instead of
614 // functional returns) assuming fecundity = 1 and 50:50 sex ratio
615 CalculateRecruitment(i_age_year, y, y);
616 this->unfished_numbers_at_age[i_age_year] =
617 fims_math::exp(this->recruitment->log_rzero[0]);
618
619 } else {
620 size_t i_agem1_yearm1 = (y - 1) * nages + (a - 1);
621 CalculateNumbersAA(i_age_year, i_agem1_yearm1, a);
622 CalculateUnfishedNumbersAA(i_age_year, i_agem1_yearm1, a);
623 }
624 CalculateBiomass(i_age_year, y, a);
625 CalculateSpawningBiomass(i_age_year, y, a);
626
627 CalculateUnfishedBiomass(i_age_year, y, a);
628 CalculateUnfishedSpawningBiomass(i_age_year, y, a);
629 }
630
631 /*
632 Here composition, total landings, and index values are calculated for
633 all years with reference data. They are not calculated for y=nyears as
634 there is this is just to get final population structure at the end of
635 the terminal year.
636 */
637 if (y < this->nyears) {
638 CalculateLandingsNumbersAA(i_age_year, y, a);
639 CalculateLandingsWeightAA(y, a);
640 CalculateLandings(y, a);
641
642 CalculateIndexNumbersAA(i_age_year, y, a);
643 CalculateIndexWeightAA(y, a);
644 CalculateIndex(i_age_year, y, a);
645 }
646 }
647 }
