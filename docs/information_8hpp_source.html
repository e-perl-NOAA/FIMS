243 void SetupPriors() {
244 for (density_components_iterator it = density_components.begin();
245 it != density_components.end(); ++it) {
246 std::shared_ptr<fims_distributions::DensityComponentBase<Type>> d =
247 (*it).second;
248 if (d->input_type == "prior") {
249 FIMS_INFO_LOG("Setup prior for distribution " + fims::to_string(d->id));
250 variable_map_iterator vmit;
251 FIMS_INFO_LOG("Link prior from distribution " + fims::to_string(d->id) +
252 " to parameter " + fims::to_string(d->key[0]));
253 d->priors.resize(d->key.size());
254 for (size_t i = 0; i < d->key.size(); i++) {
255 FIMS_INFO_LOG("Link prior from distribution " +
256 fims::to_string(d->id) + " to parameter " +
257 fims::to_string(d->key[0]));
258 vmit = this->variable_map.find(d->key[i]);
259 d->priors[i] = (*vmit).second;
260 }
261 FIMS_INFO_LOG("Prior size for distribution " + fims::to_string(d->id) +
262 "is: " + fims::to_string(d->x.size()));
263 }
264 }

760 void CreateModelingObjects(bool &valid_model) {
761 for (model_map_iterator it = this->models_map.begin();
762 it != this->models_map.end(); ++it) {
763 std::shared_ptr<fims_popdy::FisheryModelBase<Type>> &model = (*it).second;
764 std::set<uint32_t>::iterator jt;
765
766 for (jt = model->population_ids.begin();
767 jt != model->population_ids.end(); ++jt) {
768 population_iterator pt = this->populations.find((*jt));
769
770 if (pt != this->populations.end()) {
771 std::shared_ptr<fims_popdy::Population<Type>> p = (*pt).second;
772 model->populations.push_back(p);
773 for (size_t i = 0; i < p->fleets.size(); i++) {
774 model->fleets[p->fleets[i]->GetId()] = p->fleets[i];
775 }
776 } else {
777 valid_model = false;
778 FIMS_ERROR_LOG("No population object defined for model " +
779 fims::to_string(model->GetId()));
780 }
