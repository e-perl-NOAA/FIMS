103std::string finalize_fims(Rcpp::NumericVector par, Rcpp::Function fn,
104 Rcpp::Function gr) {
105 std::shared_ptr<fims_info::Information<double>> information =
106 fims_info::Information<double>::GetInstance();
107
108 std::shared_ptr<fims_model::Model<double>> model =
109 fims_model::Model<double>::GetInstance();
110 for (size_t i = 0; i < information->fixed_effects_parameters.size(); i++) {
111 *information->fixed_effects_parameters[i] = par[i];
112 }
113
114 bool reporting = model->do_tmb_reporting;
115 model->do_tmb_reporting = false;
116 model->Evaluate();
117
118 Rcpp::Function f = Rcpp::as<Rcpp::Function>(fn);
119 Rcpp::Function g = Rcpp::as<Rcpp::Function>(gr);
120 double val = Rcpp::as<double>(f(par));
121 Rcpp::NumericVector grad = Rcpp::as<Rcpp::NumericVector>(g(par));
122 double maxgc = -999;
123 for (R_xlen_t i = 0; i < grad.size(); i++) {
124 if (std::fabs(grad[i]) > maxgc) {
125 maxgc = std::fabs(grad[i]);
126 }
127 }
128
129 for (size_t i = 0; i < FIMSRcppInterfaceBase::fims_interface_objects.size();
130 i++) {
131 FIMSRcppInterfaceBase::fims_interface_objects[i]->finalize();
132 }
133 std::string ret;
134 auto now = std::chrono::system_clock::now();
135 std::time_t now_time = std::chrono::system_clock::to_time_t(now);
136 std::string ctime_no_newline = strtok(ctime(&now_time), "\n");
137 std::shared_ptr<fims_info::Information<double>> info =
138 fims_info::Information<double>::GetInstance();
139 std::stringstream ss;
140 ss << "{\n";
141 ss << "\"timestamp\": \"" << ctime_no_newline << "\",\n";
142 ss << "\"nyears\":" << info->nyears << ",\n";
143 ss << "\"nseasons\":" << info->nseasons << ",\n";
144 ss << "\"nages\":" << info->nages << ",\n";
145 ss << "\"objective_function_value\": " << val << ",\n";
146 ss << "\"max_gradient_component\": " << maxgc << ",\n";
147 ss << "\"final_gradient\": [";
148 if (grad.size() > 0) {
149 for (R_xlen_t i = 0; i < grad.size() - 1; i++) {
150 ss << grad[i] << ", ";
151 }
152 ss << grad[grad.size() - 1] << "],\n";
153 } else {
154 ss << "],";
155 }
156 ss << "\"modules\" : [\n";
157 size_t length = FIMSRcppInterfaceBase::fims_interface_objects.size();
158 if (length > 0) {
159 for (size_t i = 0; i < length - 1; i++) {
160 ss << FIMSRcppInterfaceBase::fims_interface_objects[i]->to_json()
161 << ",\n";
162 }
163
164 ss << FIMSRcppInterfaceBase::fims_interface_objects[length - 1]->to_json()
165 << "\n]\n}";
166 } else {
167 ss << "\n]\n}";
168 }
169
170 ret = fims::JsonParser::PrettyFormatJSON(ss.str());
171 model->do_tmb_reporting = reporting;
172 return ret;
173}

429std::string escapeQuotes(const std::string &input) {
430 std::string result = input;
431 std::string search = "\"";
432 std::string replace = "\\\"";
433
434 // Find each occurrence of `"` and replace it with `\"`
435 size_t pos = result.find(search);
436 while (pos != std::string::npos) {
437 result.replace(pos, search.size(), replace);
438 pos = result.find(search,
439 pos + replace.size()); // Move past the replaced position
440 }
441 return result;
442}

447void log_error(std::string log_entry) {
448 std::stringstream ss;
449 ss << "capture.output(traceback(4))";
450 SEXP expression, result;
451 ParseStatus status;
452
453 PROTECT(expression = R_ParseVector(Rf_mkString(ss.str().c_str()), 1, &status,
454 R_NilValue));
455 if (status != PARSE_OK) {
456 Rcpp::Rcout << "Error parsing expression" << std::endl;
457 UNPROTECT(1);
458 }
459 Rcpp::Rcout << "before call.";
460 PROTECT(result = Rf_eval(VECTOR_ELT(expression, 0), R_GlobalEnv));
461 Rcpp::Rcout << "after call.";
462 UNPROTECT(2);
463 std::stringstream ss_ret;
464 ss_ret << "traceback: ";
465 for (int j = 0; j < LENGTH(result); j++) {
466 std::string str(CHAR(STRING_ELT(result, j)));
467 ss_ret << escapeQuotes(str) << "\\n";
468 }
469
470 std::string ret =
471 ss_ret.str(); //"find error";//Rcpp::as<std::string>(result);
472
473 fims::FIMSLog::fims_log->error_message(log_entry, -1, "R_env", ret.c_str());
474}
