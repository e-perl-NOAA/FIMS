234 std::filesystem::path getAbsolutePathWithoutDotDot(
235 const std::filesystem::path& relativePath) {
236 std::filesystem::path absolutePath =
237 std::filesystem::absolute(relativePath);
238
239 std::filesystem::path result;
240 for (const auto& part : absolutePath) {
241 if (part == "..") {
242 if (!result.empty()) {
243 result = result.parent_path();
244 }
245 } else {
246 result /= part;
247 }
248 }
249
250 return result.generic_string();

597void WriteAtExit(int sig) {
598 std::string signal_error = "NA";
599 switch (sig) {
600 case SIGSEGV:
601 signal_error = "Invalid memory access (segmentation fault)";
602 break;
603 case SIGINT:
604 signal_error = "External interrupt, possibly initiated by the user.";
605 break;
606 case SIGABRT:
607 signal_error =
608 "Abnormal termination condition, possible call to std::abort.";
609 break;
610 case SIGFPE:
611 signal_error = "Erroneous arithmetic operation.";
612 break;
613 case SIGILL:
614 signal_error = "Invalid program image or invalid instruction";
615 break;
616 case SIGTERM:
617 signal_error = "Termination request, sent to the program.";
618 break;
619 default:
620 signal_error = "Unknown signal thrown";
621 }
622
623 FIMSLog::fims_log->error_message(signal_error, -999, "?", "?");
624
625 if (FIMSLog::fims_log->write_on_exit) {
626 std::ofstream log(FIMSLog::fims_log->get_path());
627 log << FIMSLog::fims_log->get_log();
628 log.close();
629 }
630 std::signal(sig, SIG_DFL);
631 raise(sig);
632}
